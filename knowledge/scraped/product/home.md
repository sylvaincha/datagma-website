function checkValidateEmail() {
 var emailInput = document.getElementById("email2");
 var errorMessage = document.getElementById("error-message2");
 var email = emailInput.value.trim();
 var reg = /^\w+([-+.']\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*$/;

 if (email === "") {
 errorMessage.textContent = "Please enter an email address.";
 document.querySelector(".start-btn2").disabled = true;
 } else if (reg.test(email)) {
 errorMessage.textContent = "";
 document.querySelector(".start-btn2").disabled = false;
 } else {
 errorMessage.textContent = "Please enter a valid email address.";
 document.querySelector(".start-btn2").disabled = true;
 }
 }
